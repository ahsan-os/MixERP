CREATE OR REPLACE FUNCTION hrm.get_employee_by_employee_id(_employee_id integer)
 RETURNS text
 LANGUAGE plpgsql
 STABLE
AS $function$
BEGIN
    RETURN
        employee_code || ' (' || employee_name || ')'      
    FROM hrm.employees
    WHERE hrm.employees.employee_id = $1
    AND NOT hrm.employees.deleted;    
END
$function$
